6/// Calculate 4 gradient dot products.
7///
8/// Baseline builds use table assembly because it is faster without AVX-512
9/// masks; native AVX-512 builds use the branchless hash formula.
10#[inline]
11#[must_use]
12pub fn grad_dot_4x(hashes: [usize; 4], x: f64x4, y: f64x4, z: f64x4) -> f64x4 {
13    #[cfg(target_feature = "avx512f")]
14    {
15        grad_dot_simd::<4>(hashes, x, y, z)
16    }
17
18    #[cfg(not(target_feature = "avx512f"))]
19    {
20        let mut gx = [0.0f64; 4];
21        let mut gy = [0.0f64; 4];
22        let mut gz = [0.0f64; 4];
23        for i in 0..4 {
24            let g = &GRADIENT[hashes[i] & 15];
25            gx[i] = g[0];
26            gy[i] = g[1];
27            gz[i] = g[2];
28        }
29        f64x4::from_array(gx) * x + f64x4::from_array(gy) * y + f64x4::from_array(gz) * z
30    }
31}

33/// Generic N-lane gradient dot product.
34///
35/// Evaluates Minecraft's 16-entry `GRADIENT` table branchlessly from the hash
36/// bits using the public-domain reference formula from Ken Perlin's improved
37/// noise implementation. For `hash & 15`, that formula is value-identical to
38/// indexing Minecraft's `GRADIENT` table for all 16 entries.
39///
40/// Wider SIMD uses this path because the branchless formula avoids per-lane
41/// component assembly.
42#[inline]
43#[must_use]
44pub fn grad_dot_simd<const N: usize>(
45    hashes: [usize; N],
46    x: Simd<f64, N>,
47    y: Simd<f64, N>,
48    z: Simd<f64, N>,
49) -> Simd<f64, N> {
50    let hash_lanes = Simd::<i64, N>::from_array(hashes.map(|value| (value & 15) as i64));
51    // u = h < 8 ? x : y
52    let u_component = hash_lanes.simd_lt(Simd::splat(8)).select(x, y);
53    // v = h < 4 ? y : (h == 12 || h == 14 ? x : z)
54    let v_component = hash_lanes.simd_lt(Simd::splat(4)).select(
55        y,
56        (hash_lanes.simd_eq(Simd::splat(12)) | hash_lanes.simd_eq(Simd::splat(14))).select(x, z),
57    );
58    // grad·pos = ((h & 1) == 0 ? u : -u) + ((h & 2) == 0 ? v : -v)
59    let signed_u = (hash_lanes & Simd::splat(1))
60        .simd_eq(Simd::splat(0))
61        .select(u_component, -u_component);
62    let signed_v = (hash_lanes & Simd::splat(2))
63        .simd_eq(Simd::splat(0))
64        .select(v_component, -v_component);
65    signed_u + signed_v
66}

68/// Calculate the dot product of a gradient vector and the position vector.
69#[expect(clippy::inline_always, reason = "hot-path noise primitive")]
70#[inline(always)]
71#[must_use]
72pub fn grad_dot(hash: usize, x: f64, y: f64, z: f64) -> f64 {
73    let g = &GRADIENT[hash & 15];
74    g[0] * x + g[1] * y + g[2] * z
